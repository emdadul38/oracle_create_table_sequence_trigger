// create table
CREATE TABLE "DRUG_HRM"."SUP_CLIENTS" 
(	
	"CLIENTS_ID" 	NUMBER(5,0), 
	"CLIENTS_CAT" 	NUMBER(5,0) NOT NULL, 
	"CLIENTS_NAME1" VARCHAR2(255 BYTE), 
	"ACTIVE_STATUS" CHAR(1 BYTE) DEFAULT 'Y', 
	"ENTERED_BY" 	NUMBER(5,0), 
	"ENTERED_DATE" 	DATE, 
	"UPDATE_BY" 	NUMBER(5,0), 
	"UPDATE_DATE" 	DATE, 
)

//create sequence
CREATE SEQUENCE SUP_CLIENTS_seq START WITH 1 INCREMENT BY 1;

//crate trigger
CREATE OR REPLACE TRIGGER SUP_CLIENTS_seq_tr 
BEFORE INSERT ON SUP_CLIENTS FOR EACH ROW 
WHEN (NEW.CLIENTS_ID IS NULL)

BEGIN 
	SELECT SUP_CLIENTS_seq.NEXTVAL INTO :NEW.CLIENTS_ID FROM DUAL;
END;
